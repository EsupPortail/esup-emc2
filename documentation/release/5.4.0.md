# Version 5.4.0 (XX/03/2025)

## Changements 

[CORRECTION]
* La liste des agents remontés pour les entretiens professionnels dans les structures considère maintenant la période en la date de prise de poste et la date de fin de campagne.
* Ajout d'un bandeau sur les écrans liés aux fiches de poste


## Modification de la base de données

### Sans BddAdmin

```postgresql
alter table entretienprofessionnel add convocation timestamp;

create table entretienprofessionnel_campagne_progression
(
    id                    serial
        constraint entretienprofessionnel_campagne_progression_pk
            primary key,
    campagne_id           integer                 not null
        constraint epcp_entretienprofessionnel_campagne_id_fk
            references entretienprofessionnel_campagne
            on delete cascade,
    structure_id          varchar(1024)
        constraint epcp_structure_id_fk
            references structure
            on delete cascade,
    nb_total              integer   default 0     not null,
    nb_complet            integer   default 0     not null,
    nb_autorite           integer   default 0     not null,
    nb_observation        integer   default 0     not null,
    nb_superieur          integer   default 0     not null,
    nb_planifier          integer   default 0     not null,
    nb_manquant           integer   default 0     not null,
    date_refresh          timestamp default now() not null,
    histo_creation        timestamp default now() not null,
    histo_createur_id     integer   default 0     not null
        constraint epcp_unicaen_utilisateur_user_id_fk
            references unicaen_utilisateur_user,
    histo_modification    timestamp,
    histo_modificateur_id integer
        constraint epcp_unicaen_utilisateur_user_id_fk_2
            references unicaen_utilisateur_user,
    histo_destruction     timestamp,
    histo_destructeur_id  integer
        constraint epcp_unicaen_utilisateur_user_id_fk_3
            references unicaen_utilisateur_user
);
```

### Modification des données

```postgresql
-- forçage de la date de convocation pour bricoler.
update entretienprofessionnel set convocation = date_entretien where convocation IS NULL;

-- Paramètre pour condition l'usage du bandeau sur les écrans présentant les fiches de poste
INSERT INTO unicaen_parametre_parametre(CATEGORIE_ID, CODE, LIBELLE, DESCRIPTION, VALEURS_POSSIBLES, ORDRE)
WITH d(CODE, LIBELLE, DESCRIPTION, VALEURS_POSSIBLES, ORDRE) AS (
    SELECT 'DISPLAY_BANDEAU_FICHEPOSTE', 'Affichage du bandeau sur les écrans présentant des fiches de poste', '<p>Active/désactive le bandeau présenter sur les écrans liés aux fiches de postes (associé au template <code>FICHEPOSTE_BANDEAU</code>)</p>', 'Boolean', 1000 
)
SELECT cp.id, d.CODE, d.LIBELLE, d.DESCRIPTION, d.VALEURS_POSSIBLES,  d.ORDRE
FROM d
JOIN unicaen_parametre_categorie cp ON cp.CODE = 'FICHE_POSTE';
update unicaen_parametre_parametre set valeur=true where code = 'DISPLAY_BANDEAU_FICHEPOSTE'
```
